59 void periodicTask(flexiv::Robot& robot, flexiv::Scheduler& scheduler, flexiv::Log& log,
60  flexiv::RobotStates& robotStates, const std::string& motionType,
61  const std::vector<double>& initPos)
62 {
63  // Local periodic loop counter
64  static unsigned int loopCounter = 0;
65 
66  try {
67  // Monitor fault on robot server
68  if (robot.isFault()) {
69  throw flexiv::ServerException(
70  "periodicTask: Fault occurred on robot server, exiting ...");
71  }
72 
73  // Read robot states
74  robot.getRobotStates(robotStates);
75 
76  // Robot degrees of freedom
77  size_t robotDOF = robotStates.q.size();
78 
79  // Target joint positions
80  std::vector<double> targetPos(robotDOF, 0);
81 
82  // Set target position based on motion type
83  if (motionType == "hold") {
84  targetPos = initPos;
85  } else if (motionType == "sine-sweep") {
86  for (size_t i = 0; i < robotDOF; ++i) {
87  targetPos[i]
88  = initPos[i]
89  + k_sineAmp * sin(2 * M_PI * k_sineFreq * loopCounter * k_loopPeriod);
90  }
91  } else {
92  throw flexiv::InputException(
93  "periodicTask: unknown motion type. Accepted motion types: hold, sine-sweep");
94  }
95 
96  // Run impedance control on all joints
97  std::vector<double> targetTorque(robotDOF);
98  for (size_t i = 0; i < robotDOF; ++i) {
99  targetTorque[i] = k_impedanceKp[i] * (targetPos[i] - robotStates.q[i])
100  - k_impedanceKd[i] * robotStates.dtheta[i];
101  }
102 
103  // Send target joint torque to RDK server
104  robot.streamJointTorque(targetTorque, true);
105 
106  loopCounter++;
107 
108  } catch (const flexiv::Exception& e) {
109  log.error(e.what());
110  scheduler.stop();
111  }
112 }

131  // Print description
132  log.info("Tutorial description:");
133  printDescription();
134 
135  // Type of motion specified by user
136  std::string motionType = "";
137  if (flexiv::utility::programArgsExist(argc, argv, "--hold")) {
138  log.info("Robot holding current pose");
139  motionType = "hold";
140  } else {
141  log.info("Robot running joint sine-sweep");
142  motionType = "sine-sweep";
143  }
144 
145  try {
146  // RDK Initialization
147  // =========================================================================================
148  // Instantiate robot interface
149  flexiv::Robot robot(robotIP, localIP);
150 
151  // Create data struct for storing robot states
152  flexiv::RobotStates robotStates;
153 
154  // Clear fault on robot server if any
155  if (robot.isFault()) {
156  log.warn("Fault occurred on robot server, trying to clear ...");
157  // Try to clear the fault
158  robot.clearFault();
159  std::this_thread::sleep_for(std::chrono::seconds(2));
160  // Check again
161  if (robot.isFault()) {
162  log.error("Fault cannot be cleared, exiting ...");
163  return 1;
164  }
165  log.info("Fault on robot server is cleared");
166  }
167 
168  // Enable the robot, make sure the E-stop is released before enabling
169  log.info("Enabling robot ...");
170  robot.enable();
171 
172  // Wait for the robot to become operational
173  int secondsWaited = 0;
174  while (!robot.isOperational()) {
175  std::this_thread::sleep_for(std::chrono::seconds(1));
176  if (++secondsWaited == 10) {
177  log.warn(
178  "Still waiting for robot to become operational, please check that the robot 1) "
179  "has no fault, 2) is in [Auto (remote)] mode");
180  }
181  }
182  log.info("Robot is now operational");
183 
184  // Move robot to home pose

186  robot.setMode(flexiv::Mode::NRT_PRIMITIVE_EXECUTION);
187  robot.executePrimitive("Home()");

192  }
193 
194  // Real-time Joint Torque Control
195  // =========================================================================================
196  // Switch to real-time joint torque control mode
197  robot.setMode(flexiv::Mode::RT_JOINT_TORQUE);
198 
199  // Set initial joint positions

203

207  scheduler.addTask(

209  std::ref(robotStates), std::ref(motionType), std::ref(initPos)),
210  "HP periodic", 1, scheduler.maxPriority());
211  // Start all added tasks, this is by default a blocking method
212  scheduler.start();
213 
214  } catch (const flexiv::Exception& e) {

216  return 1;
217  }
