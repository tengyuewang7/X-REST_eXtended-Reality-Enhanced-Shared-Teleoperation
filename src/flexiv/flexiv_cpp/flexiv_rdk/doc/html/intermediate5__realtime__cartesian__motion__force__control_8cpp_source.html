64 void periodicTask(flexiv::Robot& robot, flexiv::Scheduler& scheduler, flexiv::Log& log,
65  flexiv::RobotStates& robotStates, const std::vector<double>& initPose,
66  const std::string frameStr, bool enablePolish)
67 {
68  // Local periodic loop counter
69  static size_t loopCounter = 0;
70 
71  // Local flag indicating the initial contact is made
72  static bool isContacted = false;
73 
74  try {
75  // Monitor fault on robot server
76  if (robot.isFault()) {
77  throw flexiv::ServerException(
78  "periodicTask: Fault occurred on robot server, exiting ...");
79  }
80 
81  // Read robot states
82  robot.getRobotStates(robotStates);
83 
84  // Compute norm of sensed external force
85  Eigen::Vector3d extForce = {robotStates.extWrenchInBase[0], robotStates.extWrenchInBase[1],
86  robotStates.extWrenchInBase[2]};
87  double extForceNorm = extForce.norm();
88 
89  // Set sign of Fz according to reference frame to achieve a "pressing down" behavior
90  double Fz = 0.0;
91  if (frameStr == "BASE") {
92  Fz = -k_pressingForce;
93  } else if (frameStr == "TCP") {
94  Fz = k_pressingForce;
95  }
96 
97  // Initialize target vectors
98  std::vector<double> targetPose = initPose;
99  std::vector<double> targetWrench = {0.0, 0.0, Fz, 0.0, 0.0, 0.0};
100 
101  // Search for contact
102  if (!isContacted) {
103  // Send both initial pose and wrench commands, the result is force control along Z axis,
104  // and motion hold along other axes
105  robot.streamCartesianMotionForce(initPose, targetWrench);
106 
107  // Contact is made
108  if (extForceNorm > k_pressingForce) {
109  isContacted = true;
110  log.warn("Contact detected at robot TCP");
111  }
112 
113  // Skip the rest actions until contact is made
114  return;
115  }
116 
117  // Repeat the following actions in a 20-second cycle: first 15 seconds do unified
118  // motion-force control, the rest 5 seconds trigger smooth transition to pure motion control
119  if (loopCounter % (20 * k_loopFreq) == 0) {
120  // Print info at the beginning of action cycle
121  if (enablePolish) {
122  log.info("Executing polish with pressing force [N] = " + std::to_string(Fz));
123  } else {
124  log.info("Applying constant force [N] = " + std::to_string(Fz));
125  }
126 
127  } else if (loopCounter % (20 * k_loopFreq) == (15 * k_loopFreq)) {
128  log.info("Disabling force control and transiting smoothly to pure motion control");
129 
130  } else if (loopCounter % (20 * k_loopFreq) == (20 * k_loopFreq - 1)) {
131  // Reset contact flag at the end of action cycle
132  isContacted = false;
133 
134  } else if (loopCounter % (20 * k_loopFreq) < (15 * k_loopFreq)) {
135  // Simple polish action along XY plane of chosen reference frame
136  if (enablePolish) {
137  // Create motion command to sine-sweep along Y direction
138  targetPose[1]
139  = initPose[1]
140  + k_swingAmp * sin(2 * M_PI * k_swingFreq * loopCounter * k_loopPeriod);
141 
142  // Send both target pose and wrench commands, the result is force control along Z
143  // axis, and motion control along other axes
144  robot.streamCartesianMotionForce(targetPose, targetWrench);
145  }
146  // Apply constant force along Z axis of chosen reference frame
147  else {
148  // Send both initial pose and wrench commands, the result is force control along Z
149  // axis, and motion hold along other axes
150  robot.streamCartesianMotionForce(initPose, targetWrench);
151  }
152 
153  } else {
154  // By not passing in targetWrench parameter, the force control will be cancelled and
155  // transit smoothly back to pure motion control. The previously force-controlled axis
156  // will be gently pulled toward the motion target currently set for that axis. Here we
157  // use initPose for example.
158  robot.streamCartesianMotionForce(initPose);
159  }
160 
161  // Increment loop counter
162  loopCounter++;
163 
164  } catch (const flexiv::Exception& e) {
165  log.error(e.what());
166  scheduler.stop();
167  }
168 }

170 int main(int argc, char* argv[])
171 {
172  // Program Setup
173  // =============================================================================================
174  // Logger for printing message with timestamp and coloring
175  flexiv::Log log;
176 
177  // Parse parameters
178  if (argc < 3 || flexiv::utility::programArgsExistAny(argc, argv, {"-h", "--help"})) {
179  printHelp();
180  return 1;
181  }
182  // IP of the robot server
183  std::string robotIP = argv[1];
184  // IP of the workstation PC running this program
185  std::string localIP = argv[2];
186 
187  // Print description
188  log.info("Tutorial description:");
189  printDescription();
190 
191  // The reference frame to use, see Robot::streamCartesianMotionForce() for more details
192  std::string frameStr = "BASE";
193  if (flexiv::utility::programArgsExist(argc, argv, "--TCP")) {
194  log.info("Reference frame used for motion force control: robot TCP frame");
195  frameStr = "TCP";
196  } else {
197  log.info("Reference frame used for motion force control: robot base frame");
198  }
199 
200  // Whether to enable polish action
201  bool enablePolish = false;
202  if (flexiv::utility::programArgsExist(argc, argv, "--polish")) {
203  log.info("Robot will execute a polish action along XY plane");
204  enablePolish = true;
205  } else {
206  log.info("Robot will apply a constant force along Z axis");
207  }
208 
209  try {
210  // RDK Initialization
211  // =========================================================================================
212  // Instantiate robot interface
213  flexiv::Robot robot(robotIP, localIP);
214 
215  // Create data struct for storing robot states
216  flexiv::RobotStates robotStates;
217 
218  // Clear fault on robot server if any
219  if (robot.isFault()) {
220  log.warn("Fault occurred on robot server, trying to clear ...");
221  // Try to clear the fault
222  robot.clearFault();
223  std::this_thread::sleep_for(std::chrono::seconds(2));
224  // Check again
225  if (robot.isFault()) {
226  log.error("Fault cannot be cleared, exiting ...");
227  return 1;
228  }
229  log.info("Fault on robot server is cleared");
230  }
231 
232  // Enable the robot, make sure the E-stop is released before enabling
233  log.info("Enabling robot ...");
234  robot.enable();
235 
236  // Wait for the robot to become operational
237  int secondsWaited = 0;
238  while (!robot.isOperational()) {
239  std::this_thread::sleep_for(std::chrono::seconds(1));
240  if (++secondsWaited == 10) {
241  log.warn(
242  "Still waiting for robot to become operational, please check that the robot 1) "
243  "has no fault, 2) is in [Auto (remote)] mode");
244  }
245  }
246  log.info("Robot is now operational");
247 
248  // Move robot to home pose
249  log.info("Moving to home pose");
250  robot.setMode(flexiv::Mode::NRT_PRIMITIVE_EXECUTION);
251  robot.executePrimitive("Home()");
252 
253  // Wait for the primitive to finish
254  while (robot.isBusy()) {
255  std::this_thread::sleep_for(std::chrono::seconds(1));
256  }
257 
258  // Real-time Cartesian Motion-force Control
259  // =========================================================================================
260  // IMPORTANT: must zero force/torque sensor offset for accurate force/torque measurement
261  robot.executePrimitive("ZeroFTSensor()");
262 
263  // WARNING: during the process, the robot must not contact anything, otherwise the result
264  // will be inaccurate and affect following operations
265  log.warn("Zeroing force/torque sensors, make sure nothing is in contact with the robot");
266 
267  // Wait for primitive completion
268  while (robot.isBusy()) {
269  std::this_thread::sleep_for(std::chrono::seconds(1));
270  }
271  log.info("Sensor zeroing complete");
272 
273  // Get latest robot states
274  robot.getRobotStates(robotStates);
275 
276  // Set control mode and initial pose based on reference frame used
277  std::vector<double> initPose;
278  if (frameStr == "BASE") {
279  robot.setMode(flexiv::Mode::RT_CARTESIAN_MOTION_FORCE_BASE);
280  // If using base frame, directly read from robot states
281  initPose = robotStates.tcpPose;
282  } else if (frameStr == "TCP") {
283  robot.setMode(flexiv::Mode::RT_CARTESIAN_MOTION_FORCE_TCP);
284  // If using TCP frame, current TCP is at the reference frame's origin
285  initPose = {0, 0, 0, 1, 0, 0, 0};
286  } else {
287  throw flexiv::InputException("Invalid reference frame choice");
288  }
289 
290  log.info("Initial TCP pose set to [position 3x1, rotation (quaternion) 4x1]: "
291  + flexiv::utility::vec2Str(initPose));
292 
293  // Create real-time scheduler to run periodic tasks
294  flexiv::Scheduler scheduler;
295  // Add periodic task with 1ms interval and highest applicable priority
296  scheduler.addTask(
297  std::bind(periodicTask, std::ref(robot), std::ref(scheduler), std::ref(log),
298  std::ref(robotStates), std::ref(initPose), std::ref(frameStr), enablePolish),
299  "HP periodic", 1, scheduler.maxPriority());
300  // Start all added tasks, this is by default a blocking method
301  scheduler.start();
302 
303  } catch (const flexiv::Exception& e) {
304  log.error(e.what());
305  return 1;
306  }
307 
308  return 0;
309 }
