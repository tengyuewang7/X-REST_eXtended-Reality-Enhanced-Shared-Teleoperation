57 void highPriorityTask(flexiv::Robot& robot, flexiv::Scheduler& scheduler, flexiv::Log& log,
58  flexiv::RobotStates& robotStates)
59 {
60  // flag whether initial Cartesian position is set
61  static bool isInitPoseSet = false;
62 
63  // Initial Cartesian-space pose (position + rotation) of robot TCP
64  static std::vector<double> initTcpPose;
65 
66  try {
67  // Monitor fault on robot server
68  if (robot.isFault()) {
69  throw flexiv::ServerException(
70  "highPriorityTask: Fault occurred on robot server, exiting "
71  "...");
72  }
73 
74  // Read robot states
75  robot.getRobotStates(robotStates);
76 
77  // TCP movement control
78  //=====================================================================
79  // set initial TCP pose
80  if (!isInitPoseSet) {
81  // check vector size before saving
82  if (robotStates.tcpPose.size() == k_cartPoseSize) {
83  initTcpPose = robotStates.tcpPose;
84  g_currentTcpPose = initTcpPose;
85  isInitPoseSet = true;
86  }
87  }
88  // run control after initial pose is set
89  else {
90  // move along Z direction
91  g_currentTcpPose[2]
92  = initTcpPose[2]
93  + k_swingAmp * sin(2 * M_PI * k_swingFreq * g_hpLoopCounter * k_loopPeriod);
94  robot.streamCartesianMotionForce(g_currentTcpPose);
95  }
96 
97  // save data to global buffer, not using mutex to avoid interruption on
98  // RT loop from potential priority inversion
99  g_logData.tcpPose = robotStates.tcpPose;
100  g_logData.tcpForce = robotStates.extWrenchInBase;
101 
102  // increment loop counter
103  g_hpLoopCounter++;
104 
105  } catch (const flexiv::Exception& e) {
106  log.error(e.what());
107  scheduler.stop();
108  }
109 }

151 
152  // open new log file
153  csvFile.open(csvFileName);
154  if (csvFile.is_open()) {
155  log.info("Created new log file: " + csvFileName);
156  } else {
157  log.error("Failed to create log file: " + csvFileName);
158  }
159  }
160 
161  // log data to file in CSV format, avoid logging too much data otherwise
162  // the RT loop will not hold
163  if (csvFile.is_open()) {
164  // loop counter x1, TCP pose x7, TCP external force x6
165  csvFile << lpLoopCounter << ",";
166 
167  for (const auto& i : g_logData.tcpPose) {
168  csvFile << i << ",";
169  }
170 
171  for (const auto& i : g_logData.tcpForce) {
172  csvFile << i << ",";
173  }
174 
175  // end of line
176  csvFile << '\n';
177  }
178 
179  // check if the test duration has elapsed
180  if (g_hpLoopCounter > g_testDurationLoopCounts) {
181  log.info("Test duration has elapsed, saving any open log file ...");
182 
183  // close log file
184  if (csvFile.is_open()) {
185  csvFile.close();
186  log.info("Saved log file: " + csvFileName);
187  }
188 
189  // exit program
190  return;
191  }
192 
193  // increment loop counter
194  lpLoopCounter++;
195  }

221 
222  // IP of the robot server
223  std::string robotIP = argv[1];
224 
225  // IP of the workstation PC running this program
226  std::string localIP = argv[2];
227 
228  // test duration in hours
229  double testHours = std::stof(argv[3]);
230  // convert duration in hours to loop counts
231  g_testDurationLoopCounts = (uint64_t)(testHours * 3600.0 * 1000.0);
232  log.info("Test duration: " + std::to_string(testHours)
233  + " hours = " + std::to_string(g_testDurationLoopCounts) + " cycles");
234 
235  try {
236  // RDK Initialization
237  //=============================================================================
238  // Instantiate robot interface
239  flexiv::Robot robot(robotIP, localIP);
240 
241  // create data struct for storing robot states
242  flexiv::RobotStates robotStates;
243 
244  // Clear fault on robot server if any
245  if (robot.isFault()) {
246  log.warn("Fault occurred on robot server, trying to clear ...");
247  // Try to clear the fault
248  robot.clearFault();
249  std::this_thread::sleep_for(std::chrono::seconds(2));
250  // Check again
251  if (robot.isFault()) {
252  log.error("Fault cannot be cleared, exiting ...");
253  return 1;
254  }
255  log.info("Fault on robot server is cleared");
256  }
257 
258  // enable the robot, make sure the E-stop is released before enabling
259  log.info("Enabling robot ...");
260  robot.enable();
261 
262  // Wait for the robot to become operational
263  int secondsWaited = 0;
264  while (!robot.isOperational()) {
265  std::this_thread::sleep_for(std::chrono::seconds(1));
266  if (++secondsWaited == 10) {
267  log.warn(
268  "Still waiting for robot to become operational, please "
269  "check that the robot 1) has no fault, 2) is booted "
270  "into Auto mode");
271  }
272  }
273  log.info("Robot is now operational");
274 
275  // Bring Robot To Home
276  //=============================================================================
277  // set mode after robot is operational
278  robot.setMode(flexiv::Mode::NRT_PLAN_EXECUTION);
279 
280  robot.executePlan("PLAN-Home");
281 
282  // Wait fot the plan to finish
283  do {
284  std::this_thread::sleep_for(std::chrono::seconds(1));
285  } while (robot.isBusy());
286 
287  // set mode after robot is at home
288  robot.setMode(flexiv::Mode::RT_CARTESIAN_MOTION_FORCE_BASE);
289 
290  // Periodic Tasks
291  //=============================================================================
292  flexiv::Scheduler scheduler;
293  // Add periodic task with 1ms interval and highest applicable priority
294  scheduler.addTask(std::bind(highPriorityTask, std::ref(robot), std::ref(scheduler),
295  std::ref(log), std::ref(robotStates)),
296  "HP periodic", 1, scheduler.maxPriority());
297  // Start all added tasks, not blocking
298  scheduler.start(false);
299 
300  // Use std::thread for logging task without strict chronology
301  std::thread lowPriorityThread(lowPriorityTask);
302 
303  // lowPriorityThread is responsible to release blocking
304  lowPriorityThread.join();
305 
306  } catch (const flexiv::Exception& e) {
307  log.error(e.what());
308  return 1;
309  }
310 
311  return 0;
